
def calculadora():
  while True:
        print("Escolha uma das operações:")
        print("1: Soma")
        print("2: Subtração")
        print("3: Multiplicação")
        print("4: Divisão")
        print("0: Sair")

        escolha = input("Digite o número para a operação correspondente: ")

        if escolha == "0":
            print("Encerrando a calculadora.")
            break

        # 2. Verifica se a opção é válida (1 a 4)
        if escolha not in ["1", "2", "3", "4"]:
            print("Essa opção não existe")
            continue

        # 3. Entrada dos valores
        num1 = float(input("Digite o primeiro valor: "))
        num2 = float(input("Digite o segundo valor: "))

        # 4. Execução das operações
        if escolha == "1":
            print(f"Resultado: {num1 + num2}")
        elif escolha == "2":
            print(f"Resultado: {num1 - num2}")
        elif escolha == "3":
            print(f"Resultado: {num1 * num2}")
        elif escolha == "4":
            if num2 == 0:
                print("Erro: Divisão por zero não é permitido.")
            else:
                print(f"Resultado: {num1 / num2}")

# Chamando função
calculadora()
